Organizar utilizando o gemini

library(gsheet)

# Definir a URL base e os IDs das abas (gid)
url_base <- "https://docs.google.com/spreadsheets/d/1bq2N19DcZdtax2fQW9OHSGMR0X2__Z9T/edit?gid="

abas <- c(
  meudadonuvem = "1110310360",
  curve        = "1807247585",
  magnesio     = "983033137"
)

# Criar uma lista com todos os dados baixados
# O lapply percorre cada ID e aplica a função gsheet2tbl
lista_dados <- lapply(abas, function(id) gsheet2tbl(paste0(url_base, id)))

# 3. (Opcional) Se quiser os objetos soltos no ambiente como antes:
list2env(lista_dados, envir = .GlobalEnv)
