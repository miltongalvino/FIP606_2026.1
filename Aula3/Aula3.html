Corrigir a escrita de determinados valores nas colunas

#mutate: criar coluna ou modificar as existentes
#case_match: mapear valores específicos para outros

d_alla <- d_all |>                      
  mutate(trt = case_match(trt,   #mudar o nome dos valores nas colunas 
    "ctrl" ~ "Control",          #"ctrl" para "Control"
    "fung" ~ "Fungicide",        #"fung" para "Fungicide"
    .default = trt                 #garante que, se houver algum valor que não seja                                     (Control, Fungicide) permaneça intacto na                                           coluna "trt"
  )) |> 
  mutate(site = case_match (site,   
     "Uberlandia" ~ "Uberlândia",   #"Uberlandia" para "Uberlândia
     "Vicosa" ~ "Viçosa",           #"Vicosa" para "Viçosa"
     .default = site))              #garante que, se houver algum valor que não                                        seja (Viçosa e Uberlãndia) permaneça intacto na                                     coluna"site"
